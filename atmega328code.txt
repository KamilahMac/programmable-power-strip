void save_settings_to_eeprom() {
	EEPROM_write(EEPROM_INIT_FLAG, 0xAB);
	EEPROM_write_float(EEPROM_OUT1_LIMIT, outlet1_set_current_limit);
	EEPROM_write_float(EEPROM_OUT2_LIMIT, outlet2_set_current_limit);
	EEPROM_write_float(EEPROM_OUT3_LIMIT, outlet3_set_current_limit);
	EEPROM_write(EEPROM_OUT1_ENABLE, outlet1_enable);
	EEPROM_write(EEPROM_OUT2_ENABLE, outlet2_enable);
	EEPROM_write(EEPROM_OUT3_ENABLE, outlet3_enable);
}

void load_settings_from_eeprom() {
	if (EEPROM_read(EEPROM_INIT_FLAG) != 0xAB) {
		// EEPROM not initialized — write defaults
		outlet1_set_current_limit = 7.5;
		outlet2_set_current_limit = 7.5;
		outlet3_set_current_limit = 7.5;
		outlet1_enable = 0;
		outlet2_enable = 0;
		outlet3_enable = 0;

		// Save defaults — this ensures EEPROM has valid floats
		save_settings_to_eeprom();
		} else {
		// EEPROM already initialized — read floats safely
		outlet1_set_current_limit = EEPROM_read_float(EEPROM_OUT1_LIMIT);
		outlet2_set_current_limit = EEPROM_read_float(EEPROM_OUT2_LIMIT);
		outlet3_set_current_limit = EEPROM_read_float(EEPROM_OUT3_LIMIT);
		outlet1_enable = EEPROM_read(EEPROM_OUT1_ENABLE);
		outlet2_enable = EEPROM_read(EEPROM_OUT2_ENABLE);
		outlet3_enable = EEPROM_read(EEPROM_OUT3_ENABLE);
	}
}

static inline void timer1_init(void)
{
	TCCR1A = 0;
	TCCR1B = (1 << WGM12) | (1 << CS11) | (1 << CS10); // CTC, prescaler 64
	OCR1A  = 124;   // (8,000,000 / 64 / 1000) - 1 = 124  -> 1 ms
	TCNT1  = 0;
	TIMSK1 |= (1 << OCIE1A);  // enable compare match A interrupt
}


ISR(TIMER1_COMPA_vect)
{
	millis_count++;   // 1 ms tick — used for overcurrent dt

	static uint16_t ms_accum = 0;
	if (++ms_accum >= 1000) {
		ms_accum = 0;
		energy_tick = 1;   // signal main loop: 1 second has elapsed

		sec_count++;
		if (sec_count == 60) { min_count++;  sec_count = 0; }
		if (min_count == 60) { hour_count++; min_count = 0; }
	}
}

void handle_out1_trip_blink(void)
{
	const uint16_t blink_passes = BLINK_HALF_PERIOD_MS / LOOP_DELAY_MS;

	if (!out1_tripped) {
		out1_blink_state   = 0;
		out1_blink_counter = 0;
		return;
	}
	if (++out1_blink_counter >= blink_passes) {
		out1_blink_counter = 0;
		out1_blink_state  ^= 1;
		if (out1_blink_state) PORTB |=  (1 << OUTLET1_LED_PIN);
		else                  PORTB &= ~(1 << OUTLET1_LED_PIN);
	}
}

void handle_buzzer(void)
{
	if ((out1_tripped && out1_blink_state) ||
	(out2_tripped && out2_blink_state) ||
	(out3_tripped && out3_blink_state))
	{
		PORTB |= (1 << BUZZER_PIN);
	}
	else
	{
		PORTB &= ~(1 << BUZZER_PIN);
	}
}

									// ============================================================
									// RMS CURRENT
									// ============================================================
									float calculate_rms_current(uint8_t adc_channel, float *p_vopamp)
									{
										uint16_t adc_samples[SAMPLES];
										float    sum_squares = 0.0f;
										int32_t  sum = 0;
										int j;

										ADCSRA &= ~(1 << ADIE);
										ADMUX   = (ADMUX & 0xF0) | (adc_channel & 0x0F);
										_delay_us(20);

										for (j = 0; j < SAMPLES; j++) {
											ADCSRA |= (1 << ADSC);
											while (ADCSRA & (1 << ADSC));
											uint8_t adcl = ADCL;
											uint8_t adch = ADCH;
											adc_samples[j] = (adch << 8) | adcl;
											sum += adc_samples[j];
											_delay_us(SAMPLE_DELAY_US);
										}

										ADCSRA |= (1 << ADIE);

										float dc_offset = (float)sum / SAMPLES;
										for (j = 0; j < SAMPLES; j++) {
											float dev = (float)adc_samples[j] - dc_offset;
											sum_squares += dev * dev;
										}

										float rms_adc     = sqrtf(sum_squares / SAMPLES);
										float v_adc_rms   = (rms_adc * VREF) / ADC_MAX;
										float v_opamp_in  = v_adc_rms / OP_AMP_GAIN;
										float i_secondary = (v_opamp_in / BURDEN_R) * CAL_FACTOR;

										if (p_vopamp) *p_vopamp = v_opamp_in;
										return i_secondary;
									}

									// ============================================================
									// RMS VOLTAGE
									// ============================================================
									float calculate_rms_voltage(uint8_t adc_channel)
									{
										uint16_t adc_samples[SAMPLES];
										float    sum_squares = 0.0f;
										int32_t  sum = 0;
										int j;

										ADCSRA &= ~(1 << ADIE);
										ADMUX   = (ADMUX & 0xF0) | (adc_channel & 0x0F);
										_delay_us(20);

										for (j = 0; j < SAMPLES; j++) {
											ADCSRA |= (1 << ADSC);
											while (ADCSRA & (1 << ADSC));
											uint8_t adcl = ADCL;
											uint8_t adch = ADCH;
											adc_samples[j] = (adch << 8) | adcl;
											sum += adc_samples[j];
											_delay_us(SAMPLE_DELAY_US);
										}

										ADCSRA |= (1 << ADIE);

										float dc_offset = (float)sum / SAMPLES;
										for (j = 0; j < SAMPLES; j++) {
											float dev = (float)adc_samples[j] - dc_offset;
											sum_squares += dev * dev;
										}

										float rms_adc     = sqrtf(sum_squares / SAMPLES);
										float v_adc_rms   = (rms_adc * VREF) / ADC_MAX;
										float v_sensor_in = v_adc_rms / VOLT_SENSOR_GAIN;
										return v_sensor_in;
									}

												// Trip the outlet

												out1_blink_state = 0;
												out1_blink_counter = 0;

												send_string("out1 tripped ");

											} else {
											if (out1_overcurrent_detected) {

									int main(void)
									{
										usart_init(51);
										_delay_ms(100);

										DDRD |=  (1 << PIND1);
										DDRD &= ~((1 << PIND2) | (1 << PIND3));
										DDRD |=  (1 << PIND4) | (1 << PIND5) | (1 << PIND6) | (1 << PIND7);
										PORTD |= (1 << PIND2) | (1 << PIND3);
										PORTD &= ~((1 << PIND5) | (1 << PIND6) | (1 << PIND7));

										DDRB  |=  (1 << OUTLET1_LED_PIN); PORTB &= ~(1 << OUTLET1_LED_PIN);
										DDRB  |=  (1 << OUTLET2_LED_PIN); PORTB &= ~(1 << OUTLET2_LED_PIN);
										DDRD  |=  (1 << OUTLET3_LED_PIN); PORTD &= ~(1 << OUTLET3_LED_PIN);

										EICRA |= (1 << ISC11) | (1 << ISC01);
										EIMSK |= (1 << INT0)  | (1 << INT1);

										DDRC &= ~((1 << PINC0) | (1 << PINC1) | (1 << PINC2)
										| (1 << PINC3) | (1 << PINC4) | (1 << PINC5));
										ADMUX  |= (1 << REFS0);
										ADCSRA |= (1 << ADEN) | (1 << ADIE) | (1 << ADPS2) | (1 << ADPS1) | (1 << ADPS0);
										ADCSRA |= (1 << ADSC);
										while (ADCSRA & (1 << ADSC));
										ADCSRA |= (1 << ADSC);

										UCSR0B |= (1 << RXCIE0);

										/*
										enable_out1_flag = 1; disable_out1_flag = 0; out1_enabled_state = 0;
										enable_out2_flag = 1; disable_out2_flag = 0; out2_enabled_state = 0;
										enable_out3_flag = 1; disable_out3_flag = 0; out3_enabled_state = 0;
										*/

										enable_out1_flag = 0; disable_out1_flag = 0; out1_enabled_state = 0;
										enable_out2_flag = 0; disable_out2_flag = 0; out2_enabled_state = 0;
										enable_out3_flag = 0; disable_out3_flag = 0; out3_enabled_state = 0;

										DDRB |= (1 << BUZZER_PIN);
										PORTB &= ~(1 << BUZZER_PIN);   // buzzer OFF

										timer1_init();
										uint32_t last_ms = millis_count;

										//==================EEPROM START UP PRINT STATEMNET FOR ESP 32
										load_settings_from_eeprom();

										//==================EEPROM START UP PRINT STATEMENT FOR ESP32


										// Prepare the startup strings (but don't send yet)
										dtostrf(outlet1_set_current_limit, 5, 2, char_out1_curr_limit);
										dtostrf(outlet2_set_current_limit, 5, 2, char_out2_curr_limit);
										dtostrf(outlet3_set_current_limit, 5, 2, char_out3_curr_limit);

										itoa(outlet1_enable, char_out1_en_value, 10);
										itoa(outlet2_enable, char_out2_en_value, 10);
										itoa(outlet3_enable, char_out3_en_value, 10);

										// ?? Wait for ESP32 to signal it is ready ?????????????????????????
										{
											char ready_buf[16];
											uint8_t ready_idx = 0;
											uint8_t esp_ready = 0;

											// Wait up to ~30 seconds for "ESP_READY\n"
											// Each iteration: poll UART, then delay 1 ms ? ~30000 iterations max
											for (uint32_t wait = 0; wait < 30000UL && !esp_ready; wait++) {
												if (UCSR0A & (1 << RXC0)) {           // byte waiting in hardware
													char c = UDR0;
													if (c == '\n' || c == '\r') {
														ready_buf[ready_idx] = '\0';
														if (strncmp(ready_buf, "ESP_READY", 9) == 0) {
															esp_ready = 1;
														}
														ready_idx = 0;                // reset for next line
														} else {
														if (ready_idx < 15) ready_buf[ready_idx++] = c;
													}
												}
												_delay_ms(1);
											}
											// If timeout reached (no ESP connected), send anyway so standalone still works
										}

										// Now send the startup state
										strcpy(stmsg, STARTUP_OUT1_CL);
										strcat(stmsg, char_out1_curr_limit);
										strcat(stmsg, STARTUP_OUT_CL_END);
										strcat(stmsg, end_of_str);
										send_string(stmsg);

										strcpy(stmsg, STARTUP_OUT2_CL);
										strcat(stmsg, char_out2_curr_limit);
										strcat(stmsg, STARTUP_OUT_CL_END);
										strcat(stmsg, end_of_str);
										send_string(stmsg);

										strcat(stmsg, char_out3_curr_limit);
										strcat(stmsg, STARTUP_OUT_CL_END);
										strcat(stmsg, end_of_str);

										strcat(stmsg, char_out1_en_value);
										strcat(stmsg, STARTUP_OUT_EN_END);
										strcat(stmsg, end_of_str);

										strcat(stmsg, char_out3_en_value);
										strcat(stmsg, STARTUP_OUT_EN_END);
										strcat(stmsg, end_of_str);
										send_string(stmsg);

										//=============================================================================================
										//=============================================================================================
										sei();
										last_ms = millis_count;  

										while (1) {

											// measure real elapsed time since last loop iteration
										// Snapshot atomically (millis_count is 32-bit, updated in ISR)
										uint32_t now_ms;
										cli();
										now_ms = millis_count;
										sei();

										uint16_t dt_ms = (uint16_t)(now_ms - last_ms);
										last_ms = now_ms;
										
											// ============================================================
											// Current limit flags
											if (out1_curr_limit_flag) {
												out1_curr_limit_flag  = 0;

											out1_set_cur_limit = outlet1_set_current_limit;
											out2_set_cur_limit = outlet2_set_current_limit;
											out3_set_cur_limit = outlet3_set_current_limit;

											// ============================================================
											// Blink handlers
											handle_out1_trip_blink();
											handle_out2_trip_blink();
											handle_out3_trip_blink();

											handle_buzzer();

											// ============================================================
											// ADC sampling
											float acc_i1 = 0.0f, acc_v1 = 0.0f, acc_vop1 = 0.0f;
											float acc_i2 = 0.0f, acc_v2 = 0.0f, acc_vop2 = 0.0f;
											float acc_i3 = 0.0f, acc_v3 = 0.0f, acc_vop3 = 0.0f;
											float tmp_vop;
											int k;

											for (k = 0; k < AVG_READINGS; k++) {
												acc_i1   += calculate_rms_current(0, &tmp_vop);
												acc_vop1 += tmp_vop;
												acc_v1   += calculate_rms_voltage(1);

												acc_i2   += calculate_rms_current(2, &tmp_vop);
												acc_vop2 += tmp_vop;
												acc_v2   += calculate_rms_voltage(3);

												acc_i3   += calculate_rms_current(4, &tmp_vop);
												acc_vop3 += tmp_vop;
												acc_v3   += calculate_rms_voltage(5);

												_delay_ms(20);
											}

											out1_vopamp  = acc_vop1 / AVG_READINGS;
											float v1_raw = acc_v1   / AVG_READINGS;
											out1_irms    = (out1_vopamp / BURDEN_R) * 1000.0f;
											out1_vrms    = v1_raw * 609.74f;

											out2_vopamp  = acc_vop2 / AVG_READINGS;
											float v2_raw = acc_v2   / AVG_READINGS;
											out2_irms    = (out2_vopamp / BURDEN_R) * 1000.0f;
											out2_vrms    = v2_raw * 656.91f;

											out3_vopamp  = acc_vop3 / AVG_READINGS;
											float v3_raw = acc_v3   / AVG_READINGS;
											out3_irms    = (out3_vopamp / BURDEN_R) * 1000.0f;
											out3_vrms    = v3_raw * 612.88f;

											if (out1_irms < NOISE_FLOOR_A) out1_irms = 0.0f;
											if (out2_irms < NOISE_FLOOR_A) out2_irms = 0.0f;
											if (out3_irms < NOISE_FLOOR_A) out3_irms = 0.0f;

											if (out1_vrms < VOLT_NOISE_FLOOR) out1_vrms = 0.0f;
											if (out2_vrms < VOLT_NOISE_FLOOR) out2_vrms = 0.0f;
											if (out3_vrms < VOLT_NOISE_FLOOR) out3_vrms = 0.0f;

											(void)(acc_i1); (void)(acc_i2); (void)(acc_i3);

											// ============================================================
											// Format and send IRMS / VRMS / power
											/*
											dtostrf(out1_vopamp, 5, 3, out1_irms_print);
											dtostrf(v1_raw, 5, 3, out1_vrms_print);
											dtostrf(out2_vopamp, 5, 3, out2_irms_print);
											dtostrf(v2_raw, 5, 3, out2_vrms_print);
											dtostrf(out3_vopamp, 5, 3, out3_irms_print);
											dtostrf(v3_raw, 5, 3, out3_vrms_print);
											*/

											dtostrf(out1_irms, 5, 3, out1_irms_print);
											dtostrf(out1_vrms, 5, 2, out1_vrms_print);
											dtostrf(out2_irms, 5, 3, out2_irms_print);
											dtostrf(out2_vrms, 5, 2, out2_vrms_print);
											dtostrf(out3_irms, 5, 3, out3_irms_print);
											dtostrf(out3_vrms, 5, 2, out3_vrms_print);


											send_string("-- OUTLET 1 --"); send_cr_lf();
											strcpy(buffer, CURR1_ADDR);
											strcpy(str, out1_irms_print); strcat(str, CURR_ENDS); strcat(str, end_of_str);
											strcat(buffer, str); send_string(buffer); send_cr_lf();
											send_string("volt_1= "); send_string(out1_vrms_print); send_string("V"); send_cr_lf();
											send_cr_lf();

											send_string("-- OUTLET 2 --"); send_cr_lf();
											strcpy(buffer, CURR2_ADDR);
											strcpy(str, out2_irms_print); strcat(str, CURR_ENDS); strcat(str, end_of_str);
											strcat(buffer, str); send_string(buffer); send_cr_lf();
											send_string("volt_2= "); send_string(out2_vrms_print); send_string("V"); send_cr_lf();
											send_cr_lf();

											send_string("-- OUTLET 3 --"); send_cr_lf();
											strcpy(buffer, CURR3_ADDR);
											strcpy(str, out3_irms_print); strcat(str, CURR_ENDS); strcat(str, end_of_str);
											strcat(buffer, str); send_string(buffer); send_cr_lf();
											send_string("volt_3= "); send_string(out3_vrms_print); send_string("V"); send_cr_lf();
											send_cr_lf();

											out1_realtime_power_watts = out1_vrms * out1_irms ;
											out2_realtime_power_watts = out2_vrms * out2_irms ;
											out3_realtime_power_watts = out3_vrms * out3_irms ;

											if (out1_irms == 0.0f) out1_realtime_power_watts = 0.0f;
											if (out2_irms == 0.0f) out2_realtime_power_watts = 0.0f;
											if (out3_irms == 0.0f) out3_realtime_power_watts = 0.0f;
											
											
											
											if (energy_tick) {
												energy_tick = 0;   // clear flag

												// Add this second's contribution: P(W) * (1/3600) = Wh per second
												out1_energy_wh += out1_realtime_power_watts / 3600.0f;
												out2_energy_wh += out2_realtime_power_watts / 3600.0f;
												out3_energy_wh += out3_realtime_power_watts / 3600.0f;


											dtostrf(out1_energy_wh, 12, 7, act_float_engr_sen_1_print);
											dtostrf(out2_energy_wh, 12, 7, act_float_engr_sen_2_print);
											dtostrf(out3_energy_wh, 12, 7, act_float_engr_sen_3_print);

											strcpy(buffer, ENGR1_ADDR);
											strcpy(str, act_float_engr_sen_1_print); strcat(str, ENGY_ENDS); strcat(str, end_of_str);
											strcat(buffer, str); send_string(buffer); send_cr_lf();

											strcpy(buffer, ENGR2_ADDR);
											strcpy(str, act_float_engr_sen_2_print); strcat(str, ENGY_ENDS); strcat(str, end_of_str);
											strcat(buffer, str); send_string(buffer); send_cr_lf();

											strcpy(buffer, ENGR3_ADDR);
											strcpy(str, act_float_engr_sen_3_print); strcat(str, ENGY_ENDS); strcat(str, end_of_str);
											strcat(buffer, str); send_string(buffer); send_cr_lf();

											}
											// ============================================================
											// Outlet relay + LED control
											control_outlet1_pin();
											control_outlet2_pin();
											control_outlet3_pin();



											if (eeprom_dirty) {
												outlet1_enable = out1_enabled_state;
												outlet2_enable = out2_enabled_state;
												outlet3_enable = out3_enabled_state;
												save_settings_to_eeprom();
												eeprom_dirty = 0;
											}

											// ============================================================
											// Overcurrent protection — pass real elapsed ms
											check_overcurrent_out1(out1_irms, dt_ms);
											check_overcurrent_out2(out2_irms, dt_ms);
											check_overcurrent_out3(out3_irms, dt_ms);

											_delay_ms(LOOP_DELAY_MS);
										}

										return 0;

									}
